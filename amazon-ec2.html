Loading AwsSum's Amazon EC2

{% highlight javascript %} var awssum = require('awssum'); var amazon = awssum.load('amazon/amazon'); var Ec2 = awssum.load('amazon/ec2').Ec2; var ec2 = new Ec2({ 'accessKeyId' : process.env.accessKeyId, 'secretAccessKey' : process.env.secretAccessKey, 'region' : amazon.US_EAST_1 }); {% endhighlight %}
