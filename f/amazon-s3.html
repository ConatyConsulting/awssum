Loading AwsSum's Amazon S3

{% highlight javascript %} var awssum = require('awssum'); var amazon = awssum.load('amazon/amazon'); var S3 = awssum.load('amazon/s3').S3; var s3 = new S3({ 'accessKeyId' : process.env.accessKeyId, 'secretAccessKey' : process.env.secretAccessKey, 'region' : amazon.US_EAST_1 }); {% endhighlight %}
