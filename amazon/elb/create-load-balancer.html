Create a loadbalancer

This example shows how to create a loadbalancer.

Example

var params = {
    LoadBalancerName : 'test',
    Listeners : [{
      LoadBalancerPort : 80,
      Protocol : 'http',
      InstancePort : 3000
    }],
    AvailabilityZones: [amazon.EU_WEST_1+'a']
}
elb.CreateLoadBalancer(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
