Create a new hosted zone in Route53

This example shows how to make a new hosted zone (/domain).

Example

var params = {
    Name            : 'newdomain.com',
    CallerReference : 'unique_string_to_ensure_action_will_not_be_called_twice_1234',
    Comment         : 'Created a new hosted zone.'
}

r53.CreateHostedZone(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
