Set identity feedback forwarding to enabled/disabled

This example shows how to enable/disable feedback forwarding on a identity (domain/email address). It is important for bounce and complaint notifications.

Example

var params1 = {
    Identity          : 'email@domain.com',
    ForwardingEnabled : true,
};
ses.SetIdentityFeedbackForwardingEnabled(params1, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

    Identity          : 'email@domain.com',
    ForwardingEnabled : false,
};
