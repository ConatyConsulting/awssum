Examples

Set an identity notification (SNS) topic

This example shows how to set the notification (SNS) topic per identity to publish bounce and complaint notifications to.

Example

var bounce = {
    Identity         : 'email@domain.com',
    NotificationType : 'Bounce',
    SnsTopic         : 'arn:aws:sns:us-east-1:123456789012:exampleTopic',
};
ses.SetIdentityNotificationTopic(bounce, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

var complaint = {
    Identity         : 'domain.com',
    NotificationType : 'Complaint',
    SnsTopic         : 'arn:aws:sns:us-east-1:123456789012:exampleTopic'
};
ses.SetIdentityNotificationTopic(complaint, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
