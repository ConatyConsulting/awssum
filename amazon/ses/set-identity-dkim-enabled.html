Param Name Required Type Notes
Action required form
DkimEnabled required form

Examples

Set identity DKIM to enabled/disabled

This example shows how to enable/disable DKIM on a identity (domain/email address).

Example

var params1 = {
    Identity    : 'email@domain.com',
    DkimEnabled : true,
};
ses.SetIdentityDkimEnabled(params1, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

var params2 = {
    Identity    : 'email@domain.com',
    DkimEnabled : false,
};
ses.SetIdentityDkimEnabled(params2, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
