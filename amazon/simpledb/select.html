Examples

List all attributes from a domain

This example shows how to list all attributes of the items stored in a domain.

Example

var params = {
    SelectExpression : 'SELECT * FROM an_unique_string'
};
sdb.Select(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

List a specific attribute from a domain

This example shows how to list a specific attribute of the items stored in a domain.

Example

var params = {
    SelectExpression : 'SELECT id,name FROM an_unique_string'
};
sdb.Select(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

List the most recent data from a domain

This example shows how to list the most recent data from a domain.

Example

var params = {
    SelectExpression : 'SELECT * FROM an_unique_string',
    ConsistentRead   : true
};
sdb.Select(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
