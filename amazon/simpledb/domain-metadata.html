Show metadata for a specific domain

This example shows how to show metadata for a specific domain.

Example

var params = {
    DomainName : 'test-domain'
};
sdb.DomainMetadata(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
