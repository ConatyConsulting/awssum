List all queues

This example shows how to list all queues

Example

sqs.ListQueues({}, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

List all queues with a specific prefix

This example shows how to list all queues starting with a specific prefix ("_dev")

Example

var params = {
    QueueNamePrefix : '_dev'
}
sqs.ListQueues(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
