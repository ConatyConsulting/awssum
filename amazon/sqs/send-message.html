Send a message onto a SQS queue

This example shows how to send a message onto a SQS queue to process later in order.

Example

var custom = {
    aCertainType       : 'csv',
    orAnotherProperty  : 'like a string',
    yetAnotherProperty : true,
};
var params = {
    QueueName   : 'yourQueueName',
    MessageBody : JSON.stringify(custom),
};
sqs.SendMessage(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
