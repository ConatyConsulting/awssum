PutMetricData (Open docs on AWS)

Params

Param Name Required Type Notes
Action required param
MetricData required param-data
Namespace required param

Examples

Publish metric data points to Cloudwatch

This example shows how to publish metric data points to Cloudwatch.

Example

var params = {
    MetricData : [
    {
        MetricName  : 'Counter',
        Unit        : 'Count',
        Value       : 1,
        Timestamp   : new Date().toISOString(8601),
        Dimensions  : [
            { Name : 'InstanceId',   Value : 'i-aaba32d5' },
            { Name : 'InstanceType', Value : 'm1.micro' }
        ]
    }],
    Namespace : 'test'
}
cw.PutMetricData(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

This example shows how to publish metric (time) data points to Cloudwatch.

Example

var params = {
    MetricData : [
    {

        Unit            : 'Milliseconds',
        StatisticValues : {
            Minimum: 3,
            Maximum: 1500,
            Sum: 3000,
            SampleCount: 10
        }
    }],
    Namespace : 'test'
}
cw.PutMetricData(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
