Get session token

This example shows how to get default temporary security credentials.

Example

sqs.GetSessionToken({}, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

Get session token for 14 hours

This example shows how to get temporary security credentials that last 14 hours. Your temporary credentials can last between 1 and 36 hours and are default set to 12 hours.

Example

var params = {
    DurationSeconds : 50400
}
sqs.GetSessionToken(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});

Get session token through your MFA-device

This example shows how to get temporary security credentials through an connected MFA-device.

Example

var params = {
    SerialNumber : 'arn:aws:iam::123456789012:mfa/user',
    TokenCode    : '123456'
}
/* OR
var params = {
    SerialNumber : 'GAHT12345678',
    TokenCode    : '123456'
}
*/
sqs.GetSessionToken(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
